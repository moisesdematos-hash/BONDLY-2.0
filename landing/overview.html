class BondlyChatScreen extends ConsumerStatefulWidget {
  // Chat com análise de sentimento em tempo real
  // Cada mensagem tem um sentiment_score (0.0–1.0)

  Widget _buildMessageList() {
    return ListView.builder(
      reverse: true,
      itemBuilder: (ctx, i) {
        final msg = chatState.messages[i];
        final isMe = msg['sender_id'] == userId;
        return _ChatBubble(
          message: msg, isMe: isMe);
      },
    );
  }
}

  // Bolhas com bordas assimétricas
  // Indicador de sentimento por mensagem
  // Tips de IA inline quando relevante

  sentiment > 0.7 → 😊 Verde (positivo)
  sentiment < 0.4 → 😟 Vermelho (alerta)
  
  // Dica contextual da IA:
  if (message['meta']?['tip'] != null)
    Text('💡 \${tip}',
      style: TextStyle(
        color: Color(0xFF2DD4BF)));
}

class GardenWidget extends ConsumerWidget {
  // Visual dinâmico baseado em level + health

  Widget _buildGardenVisual(level, health) {
    if (health < 40) {
      icon = Icons.eco_outlined;  // Murcho
      color = Colors.brown;
    } else if (level < 3) {
      icon = Icons.eco;           // Broto
      color = Colors.greenAccent;
    } else if (level < 7) {
      icon = Icons.local_florist; // Flor
      color = Colors.pinkAccent;
    } else {
      icon = Icons.auto_awesome;  // Pleno
      color = Colors.amberAccent;
    }

    return TweenAnimationBuilder<double>(
      tween: Tween(begin: 0.8, end: 1.0),
      duration: Duration(seconds: 2),
      curve: Curves.easeInOutSine,
      child: Container(
        boxShadow: [BoxShadow(
          color: color.withOpacity(0.2),
          blurRadius: 30, spreadRadius: 10)],
        child: Icon(icon, size: size, color: color),
      ),
    );
  }

  Widget _buildHealthIndicator(health) {
    // Verde > 70%, Amarelo > 30%, Vermelho < 30%
    Color color = health > 70
      ? Color(0xFF2DD4BF)
      : health > 30 ? Colors.amberAccent
      : Colors.redAccent;
  }

  Widget _buildXpBar(xp, max) {
    return LinearProgressIndicator(
      value: (xp / max).clamp(0.0, 1.0),
      color: Color(0xFF6366F1),
    );
  }
}

class DatePlannerScreen extends ConsumerStatefulWidget {
  // Sugestões de dates geradas por IA
  // Baseadas no humor e métricas do casal

  @override
  void initState() {
    ref.read(dateProvider.notifier)
      .fetchSuggestions(relationshipId);
  }

  Widget _buildDateCard(suggestion) {
    final isCasa = suggestion['tipo'] == 'casa';
    final isPremium = suggestion['premium'];

    return Container(
      decoration: BondlyTheme.glassDecoration(),
      child: Column([
        // Badge: 🏠 EM CASA ou 🚗 RUA
        Row([
          Text(isCasa ? '🏠 EM CASA' : '🚗 RUA'),
          if (isPremium)
            Icon(Icons.star, color: amber),
        ]),
        Text(suggestion['titulo']),
        Text(suggestion['descricao']),
        // Seção "Por que sugerimos?"
        Container(child: Row([
          Icon(Icons.auto_fix_high),
          Text('POR QUE SUGERIMOS?'),
          Text(suggestion['por_que']),
        ])),
      ]),
    );
  }

  // Loading state com texto animado:
  // "Bondly IA está criando momentos..."
}
